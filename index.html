Google Sheets y cuenta
📊 Conectar Google Sheets
Todos los registros se sincronizan automaticamente con tu hoja de calculo compartida con el equipo.
  1. Crea una Google Sheet llamada CRM-ICSSA
  2. Ve a Extensiones > Apps Script y pega el codigo de abajo
  3. Implementa como Aplicacion web > acceso para Cualquiera
  4. Copia la URL generada y pegala en el campo de abajo
function doPost(e) { const d = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName("Servicios"); if (!sh) { sh = ss.insertSheet("Servicios"); sh.appendRow(["ID","Fecha","Cliente","Tel","Email", "Direccion","Tipo","Estado","Tecnico","Capturo", "Notas","Maps","Mantenimiento","Marca","Modelo", "Serie","Capacidad","Tienda","ObsEquipo", "TipoCobro","ManoObra","Refacciones","Total"]); } sh.appendRow([d.id,d.fecha,d.cliente,d.tel,d.email, d.direccion,d.tipo,d.estado,d.tecnico,d.capturo, d.notas,d.maps,d.mantenimiento,d.marca,d.modelo, d.serie,d.capacidad,d.tienda,d.equipo_obs, d.tipo_cobro,d.mano_obra,d.refacciones_txt,d.total]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
